class furniture(object):
    # 构造方法
    def __init__(self, name, area):
        self.name = name
        self.area = area

    # toString
    def __str__(self):
        return f'家具名字:{self.name},家具大小:{self.area}'


class house(object):
    # 构造方法
    def __init__(self, address, area):
        self.address = address
        self.area = area
        # 房子里的家具列表
        self.list_furniture = []
        # 房子剩余的大小
        self.remain_furniture = area

    # 添加家具
    def add_furniture(self, furniture):
        # 有空间
        if self.remain_furniture > furniture.area:
            # 添加到列表
            self.list_furniture.append(furniture)
            print(f'家具{furniture.name}添加成功')
            # 房子大小减小
            self.remain_furniture -= furniture.area
        else:
            print('房子不够大')

    def __str__(self):
        # 获取所有家具名字
        furniture_list = [obj.name for obj in self.list_furniture]
        return f'房子地址{self.address},房子现在大小{self.remain_furniture},房子原来大小{self.area},房子的家具{",".join(furniture_list)}'


f = furniture('椅子', 20)
h = house('广东', 100)
h.add_furniture(f)
h.add_furniture(f)
print(f)
print(h)

    # 父类(基类):Animal
class Animal(object):

# 子类(派生类):Dog
class Dog(Animal):
    # 重写父类的方法:因为父类的方法不满足需求,需要扩展
    def play(self):

    # 子类调用父类的方法
    def see_host(self):
        # 方法一
        super().play()
        # 方法二:说明子类是谁
        super(Dog, self).play()

#student_manage_system
import student as s


class StudentManagerSystem(object):
    # 构造方法，一个字典：键为学号，值为学生信息
    def __init__(self):
        self.stu_dict = {}

    # 预加载
    def load_init(self):
        student = open('student.txt', 'r', encoding='utf-8')
        # stuAll:['zhangsan,233,nan,101\n', 'wangwu,21,nan,103\n']
        stuAll = student.readlines()
        for stu in stuAll:
            # 把换行符(\n)去掉,因为去空白，相当于把换行多余的空白去掉,stu_strip：字符串
            stu_strip = stu.strip()
            # 返回字符串中单词的列表
            stuInfo = stu_strip.split(',')
            self.stu_dict[stuInfo[3]] = s.Student(*stuInfo)
        pass

    # 添加学生
    def addStudent(self):
        stu_id = input('请输入学生的学号：')
        name = input('请输入学生姓名：')
        age = int(input('请输入学生年龄：'))
        gender = input('请输入学生性别：')
        if stu_id in self.stu_dict:
            print("该学生已存在")
        else:
            stu = s.Student(name, age, gender, stu_id)
            self.stu_dict[stu_id] = stu
        pass

    # 删除学生
    def deleteStudent(self):
        stu_id = input('请输入学生的学号：')
        # 根据学生学号删除学生
        del self.stu_dict[stu_id]
        pass

    # 修改学生
    def modifyStudent(self):
        stu_id = input('请输入学生的学号：')
        # 根据学生学号删除学生
        if stu_id in self.stu_dict:
            age = int(input('请输入修改学生的年龄：'))
            # 根据键获取值（学生）
            stu = self.stu_dict.get(stu_id)
            stu.age = age
        pass

    # 查询一个学生信息
    def selectOneStudent(self):
        stu_id = input('请输入学生的学号：')
        if stu_id in self.stu_dict:
            # 根据键获取值（学生）
            stu = self.stu_dict.get(stu_id)
            print(stu)
        pass

    # 查询所有学生信息
    def selectAllStudent(self):
        for stu in self.stu_dict.values():
            print(stu)
        pass

    # 保存学生信息
    def save(self):
        student = open('student.txt', 'w', encoding='utf-8')
        for stu in self.stu_dict.values():
            student.write(str(stu) + '\n')
        student.close()
        pass

    # 主界面
    @staticmethod
    def show_menu():
        print("1.添加学生")
        print("2.删除学生")
        print("3.修改学生")
        print("4.查询单个学生")
        print("5.查询所有学生")
        print("6.退出系统")
        pass

    def start(self):
        # 预加载
        self.load_init()
        # 主界面
        while (True):
            self.show_menu()
            ans = int(input("请选择要进行的编号："))
            if ans == 1:
                self.addStudent()
            elif ans == 2:
                self.deleteStudent() 
            elif ans == 3:
                self.modifyStudent()
            elif ans == 4:
                self.selectOneStudent()
            elif ans == 5:
                self.selectAllStudent()
            elif ans == 6:
                print('退出系统')
                self.save()
                break
            else:
                print('输入错误，请重新输入！！！')
                continue
